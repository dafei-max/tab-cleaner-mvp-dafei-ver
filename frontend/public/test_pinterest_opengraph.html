Pinterest OpenGraph 本地抓取测试

打开 Chrome DevTools Console，然后运行：

// 方法 1：直接测试函数
window.__TAB_CLEANER_GET_OPENGRAPH()

// 方法 2：等待页面加载完成
window.__TAB_CLEANER_GET_OPENGRAPH(true).then(result => {
  console.log('OpenGraph 结果:', result);
  console.log('图片 URL:', result.image);
});
